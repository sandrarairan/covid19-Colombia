SIR <- function(time, state, parameters) {
  par <- as.list(c(state, parameters))
  with(par, {
    dS <- -beta/N * I * S
    dI <- beta/N * I * S - gamma * I
    dR <- gamma * I
    list(c(dS, dI, dR))
    })
}

#Valores iniciales para el modelo SIR  N, S, I and R
# N poblacion de colombia
# S nÃºmero de individuos susceptibles al virus
# I infectados
# R Recuperados 
N <- 48258494
init <- c(
  S = N - Infected[1],
  I = Infected[1],
  R = 0
)

#Luego necesitamos definir una funciÃ³n para calcular el RSS , dado un conjunto de valores para Î² y Î³

# Î² La tasa de infecciÃ³n, que controla la transiciÃ³n entre S e I
# Î³ La tasa de eliminaciÃ³n o recuperaciÃ³n, que controla la transiciÃ³n entre I y R

# definir una funciÃ³n para calcular la suma residual de cuadrados
# (RSS), pasando los parÃ¡metros beta y gamma
RSS <- function(parameters) {
  names(parameters) <- c("beta", "gamma")
  out <- ode(y = init, times = Day, func = SIR, parms = parameters)
  fit <- out[, 3]
  sum((Infected - fit)^2)
}


# encontrar los valores "beta", "gamma

# install.packages("deSolve")
library(deSolve)

Opt <- optim(c(0.5, 0.5),
  RSS,
  method = "L-BFGS-B",
  lower = c(0, 0),
  upper = c(1, 1)
)

**Modelo SIR (Susceptible, Infectada y Recuperada)**

SIR <- function(time, state, parameters) {
  par <- as.list(c(state, parameters))
  with(par, {
    dS <- -beta/N * I * S
    dI <- beta/N * I * S - gamma * I
    dR <- gamma * I
    list(c(dS, dI, dR))
    })
}
