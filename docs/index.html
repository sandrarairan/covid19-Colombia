N <- 48258494
SIR <- function(time, state, parameters) {
  par <- as.list(c(state, parameters))
  with(par, {
    dS <- -beta/N * I * S
    dI <- beta/N * I * S - gamma * I
    dR <- gamma * I
    list(c(dS, dI, dR))
    })
}

# Cree un vector de día incremental de los dias que hemos tenido el Virus

Day <- 1:(length(Infected))

#Valores iniciales para el modelo SIR  N, S, I and R
# N poblacion de colombia
# S número de individuos susceptibles al virus
# I infectados
# R Recuperados 
N <- 48258494
init <- c(
  S = N - Infected[1],
  I = Infected[1],
  R = 0
)

#Luego necesitamos definir una función para calcular el RSS , dado un conjunto de valores para β y γ

# β La tasa de infección, que controla la transición entre S e I
# γ La tasa de eliminación o recuperación, que controla la transición entre I y R

# definir una función para calcular la suma residual de cuadrados
# (RSS), pasando los parámetros beta y gamma
RSS <- function(parameters) {
  names(parameters) <- c("beta", "gamma")
  out <- ode(y = init, times = Day, func = SIR, parms = parameters)
  fit <- out[, 3]
  sum((Infected - fit)^2)
}


# encontrar los valores "beta", "gamma

# install.packages("deSolve")
library(deSolve)

Opt <- optim(c(0.5, 0.5),
  RSS,
  method = "L-BFGS-B",
  lower = c(0, 0),
  upper = c(1, 1)
)

**Modelo SIR (Susceptible, Infectada y Recuperada)**

SIR <- function(time, state, parameters) {
  par <- as.list(c(state, parameters))
  with(par, {
    dS <- -beta/N * I * S
    dI <- beta/N * I * S - gamma * I
    dR <- gamma * I
    list(c(dS, dI, dR))
    })
}
